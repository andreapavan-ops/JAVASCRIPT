1.7 — Metodi principali

In JavaScript quasi tutto è un oggetto e gli oggetti hanno metodi: funzioni "attaccate" a un valore che lo manipolano o restituiscono informazioni su di esso. Si chiamano con la notazione punto: valore.metodo().

Regola #1 — Le stringhe sono immutabili.
I metodi sulle stringhe non modificano la stringa originale: restituiscono sempre una nuova stringa. Devi salvare il risultato in una variabile se vuoi usarlo.

Regola #2 — length è una proprietà, non un metodo.
Si scrive str.length senza parentesi, non str.length(). Se scrivi le parentesi ottieni un errore: TypeError: str.length is not a function. 
let frase = "  Ciao Mondo!  ";

// .trim() — rimuove gli spazi iniziali e finali
let pulita = frase.trim();
console.log(pulita);                 // "Ciao Mondo!"

// .length — numero di caratteri (dopo il trim)
console.log(pulita.length);          // 11

// .toUpperCase() / .toLowerCase()
console.log(pulita.toUpperCase());   // "CIAO MONDO!"
console.log(pulita.toLowerCase());   // "ciao mondo!"

// REGOLA #1 in pratica: la stringa originale è intatta
console.log(pulita);                 // "Ciao Mondo!"  ← invariata

// .includes() — ricerca booleana
console.log(pulita.includes("Mondo")); // true
console.log(pulita.includes("ciao")); // false  ← case sensitive!

// .startsWith() / .endsWith()
console.log(pulita.startsWith("Ciao")); // true
console.log(pulita.endsWith("!"));      // true

// .indexOf() — posizione (0-based), -1 se non trovato
console.log(pulita.indexOf("Mondo"));  // 5
console.log(pulita.indexOf("xyz"));    // -1

// .slice(start, end) — estrae sottostringa, end escluso
console.log(pulita.slice(5, 10));    // "Mondo"
console.log(pulita.slice(-6));       // "ondo!"  ← indice negativo: parte dalla fine

// .replace() — sostituisce solo la PRIMA occorrenza
let ripetuta = "ha ha ha";
console.log(ripetuta.replace("ha", "no"));    // "no ha ha"

// .replaceAll() — sostituisce TUTTE le occorrenze
console.log(ripetuta.replaceAll("ha", "no")); // "no no no"

// .split() — divide in array
let csv = "mario,luigi,peach";
console.log(csv.split(","));         // ["mario", "luigi", "peach"]
console.log("ciao".split(""));    // ["c","i","a","o"]  ← split su stringa vuota

// .repeat() — ripete la stringa
console.log("*".repeat(5));       // "*****"

// .padStart() / .padEnd() — utile per formattare output
console.log("7".padStart(3, "0")); // "007"
console.log("hi".padEnd(5, "."));  // "hi..."

2. Metodi sui numeri e oggetto Math

Regola #3 — Math è un oggetto globale con metodi statici.
Non si istanzia: si usa direttamente come Math.round(), Math.max(), ecc. È simile al modulo math di Python, ma senza import.
Regola #4 — Math.random() restituisce un numero tra 0 (incluso) e 1 (escluso).
Per ottenere un intero casuale tra min e max (inclusi):
Math.floor(Math.random() * (max - min + 1)) + min 
// ----- Metodi sull'istanza Number -----

let pi = 3.14159265;

// .toFixed(n) — arrotonda a n decimali, restituisce una STRINGA
console.log(pi.toFixed(2));          // "3.14"  ← attenzione: è una stringa!
console.log(typeof pi.toFixed(2));   // "string"

// .toString() — converte il numero in stringa
console.log((255).toString());       // "255"
console.log((255).toString(16));     // "ff"  ← base 16 (esadecimale)
console.log((8).toString(2));        // "1000" ← base 2 (binario)

// ----- Metodi statici su Number -----

// Number.isInteger() — controlla se il numero è intero
console.log(Number.isInteger(42));   // true
console.log(Number.isInteger(3.14)); // false

// Number.isNaN() — versione affidabile di isNaN globale
console.log(Number.isNaN(NaN));      // true
console.log(Number.isNaN("ciao"));   // false  ← NON converte prima, più sicuro
console.log(isNaN("ciao"));          // true   ← isNaN globale converte prima: pericoloso!

// Number.isFinite() — controlla che non sia Infinity o NaN
console.log(Number.isFinite(42));    // true
console.log(Number.isFinite(Infinity)); // false

// ----- Oggetto Math -----

// Math.round() — arrotonda all'intero più vicino
console.log(Math.round(4.5));        // 5
console.log(Math.round(4.4));        // 4

// Math.floor() — arrotonda VERSO IL BASSO (tronca)
console.log(Math.floor(4.9));        // 4
console.log(Math.floor(-4.1));       // -5  ← attenzione ai negativi!

// Math.ceil() — arrotonda VERSO L'ALTO
console.log(Math.ceil(4.1));         // 5
console.log(Math.ceil(-4.9));        // -4  ← attenzione ai negativi!

// Math.abs() — valore assoluto
console.log(Math.abs(-7));           // 7

// Math.max() / Math.min() — massimo e minimo tra argomenti
console.log(Math.max(3, 1, 4, 1, 5, 9)); // 9
console.log(Math.min(3, 1, 4, 1, 5, 9)); // 1

// Math.pow() — potenza (equivalente a **)
console.log(Math.pow(2, 10));        // 1024
console.log(2 ** 10);                // 1024  ← sintassi moderna equivalente

// Math.sqrt() — radice quadrata
console.log(Math.sqrt(16));          // 4

// Math.random() — numero casuale [0, 1)
console.log(Math.random());          // es. 0.7341...  (varia ad ogni esecuzione)

// REGOLA #4 in pratica: intero casuale da 1 a 6 (dado)
let dado = Math.floor(Math.random() * 6) + 1;
console.log("Dado:", dado);          // numero da 1 a 6

// Math.PI e Math.E — costanti matematiche
console.log(Math.PI);                // 3.141592653589793
console.log(Math.E);                 // 2.718281828459045

3. Metodi sugli array

Regola #5 — Alcuni metodi modificano l'array originale (mutating), altri restituiscono un nuovo array senza modificarlo (non-mutating).
Conoscere questa distinzione evita bug difficili da trovare.

.unshift(el)Aggiunge in testaSì
.shift()Rimuove e restituisce il primoSì
.splice(i, n)Rimuove n elementi da indice iSì
.reverse()Inverte l'ordineSì
.sort()Ordina (default: lessicografico)Sì

Regola #6 — .sort() ordina come stringhe per default.
[10, 9, 2].sort() restituisce [10, 2, 9] perché confronta "10" < "2" lessicograficamente. Per ordinare numeri correttamente usa il comparatore: .sort((a, b) => a - b). 
let frutti = ["mela", "banana", "ciliegia"];

// .push() — aggiunge in fondo, restituisce la nuova lunghezza
let nuovaLen = frutti.push("dattero");
console.log(frutti);    // ["mela","banana","ciliegia","dattero"]
console.log(nuovaLen);  // 4

// .pop() — rimuove l'ultimo e lo restituisce
let ultimo = frutti.pop();
console.log(ultimo);    // "dattero"
console.log(frutti);    // ["mela","banana","ciliegia"]

// .unshift() — aggiunge in testa
frutti.unshift("albicocca");
console.log(frutti);    // ["albicocca","mela","banana","ciliegia"]

// .shift() — rimuove il primo e lo restituisce
let primo = frutti.shift();
console.log(primo);     // "albicocca"
console.log(frutti);    // ["mela","banana","ciliegia"]

// .includes() — ricerca booleana
console.log(frutti.includes("banana")); // true
console.log(frutti.includes("kiwi"));   // false

// .indexOf() — indice dell'elemento (-1 se non trovato)
console.log(frutti.indexOf("banana"));  // 1
console.log(frutti.indexOf("kiwi"));    // -1

// .slice() — estrae senza modificare (come le stringhe)
let copia = frutti.slice(0, 2);
console.log(copia);     // ["mela","banana"]
console.log(frutti);    // ["mela","banana","ciliegia"]  ← invariato

// .join() — unisce in stringa (inverso di split)
console.log(frutti.join(" - "));       // "mela - banana - ciliegia"
console.log(frutti.join(""));         // "melebananaciliegia"

// .reverse() — inverte L'ORIGINALE
let numeri = [1, 2, 3, 4, 5];
numeri.reverse();
console.log(numeri);    // [5,4,3,2,1]  ← originale modificato!

// .sort() con comparatore numerico (REGOLA #6)
let disordinati = [10, 9, 2, 100, 21];
console.log(disordinati.sort());              // [10,100,2,21,9]  ← SBAGLIATO!
console.log(disordinati.sort((a, b) => a - b)); // [2,9,10,21,100] ← CORRETTO

// .map() — trasforma ogni elemento in un nuovo array
let prezzi = [10, 20, 30];
let scontati = prezzi.map(p => p * 0.9);
console.log(scontati);  // [9, 18, 27]
console.log(prezzi);    // [10, 20, 30]  ← originale invariato

// .filter() — restituisce elementi che soddisfano la condizione
let voti = [4, 7, 5, 8, 3, 9, 6];
let promossi = voti.filter(v => v >= 6);
console.log(promossi);  // [7, 8, 9, 6]

// .find() — restituisce il PRIMO elemento che soddisfa (undefined se nessuno)
let primoAlto = voti.find(v => v >= 8);
console.log(primoAlto); // 8

// .forEach() — esegue una funzione per ogni elemento (non restituisce niente)
["a", "b", "c"].forEach((el, i) => {
    console.log(i, el);  // 0 "a" | 1 "b" | 2 "c"
});

// .reduce() — accumula un risultato su tutti gli elementi
let somma = [1, 2, 3, 4, 5].reduce((acc, el) => acc + el, 0);
console.log(somma);     // 15

let prodotto = [1, 2, 3, 4, 5].reduce((acc, el) => acc * el, 1);
console.log(prodotto);  // 120
💡 Concatenare metodi (method chaining): 
let risultato = [1, 2, 3, 4, 5, 6]
    .filter(n => n % 2 === 0)   // [2, 4, 6]   ← solo i pari
    .map(n => n * n)             // [4, 16, 36] ← elevati al quadrato
    .join(", ");                 // "4, 16, 36" ← uniti in stringa

console.log(risultato);         // "4, 16, 36"
